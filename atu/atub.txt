import automated.tests.annotations.SkipTesting;
import automated.tests.cli.AutomatedTestsUnits;
import automated.tests.cli.TestsUnitairesAutomatises;
import automated.tests.types.CodeGenerationsTypes;

/**
 * Génération des tests unitaires automatisés du projet Java
 */
@SkipTesting
public class JavaAutomatedTestsBuilder {

    public static void main(String[] args) {
        boolean french = true; // Indicateur de générer les commentaires en français dans les tests unitaires générés
        // boolean french = false; // Indicateur de générer les commentaires en anglais dans les tests unitaires générés

        CodeGenerationsTypes codeGenerationType = CodeGenerationsTypes.AUTO_UPDATE; // Régénère uniquement les cas de tests Junit des classes source Java modifiées
        // CodeGenerationsTypes codeGenerationType = CodeGenerationsTypes.FULL; // Supprime tout et régénère tous les cas de test Junit de toutes les classes source Java
        // CodeGenerationsTypes codeGenerationType = CodeGenerationsTypes.REFRESH; // Régénère tous les cas de tests Junit, mais sans supprimer tout au début

        boolean debug = false; // Exécution de l'outil sans mode de débogage pour les tests effectués en langage Groovy lors de l'analyse
        // boolean debug = true; // Exécute en mode débogage en effectuant un seul test à la fois en langage Groovy. Utile pour suivre les problèmes, mais ralentit la génération de cas de tests Junit !!!

        String packageName = null; // Nom du package de base Java du répertoire de sortie pour les cas de test Junit générés. Par défaut: junit

        if (french) {
            TestsUnitairesAutomatises.javaTestsUnitairesAutomatises(packageName, codeGenerationType, debug);
        } else {
            AutomatedTestsUnits.javaAutomatedTestsUnits(packageName, codeGenerationType, debug);
        }
    }

}
